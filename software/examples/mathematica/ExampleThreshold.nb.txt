Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="amb"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
dir=NETNew["Tinkerforge.BrickletDistanceIR",uid,ipcon]
ipcon@Connect[host,port]

(* Get threshold callbacks with a debounce time of 10 seconds (10000ms) *)
dir@SetDebouncePeriod[10000]

(* Callback for distance smaller than 20 cm *)
DistanceReachedCB[sender_,distance_]:=
 Print["Distance is smaller than 20 cm: "<>ToString[N[Quantity[distance,"cm/10"]]]]
AddEventHandler[dir@DistanceReached,DistanceReachedCB]

(* Configure threshold for "smaller than 20 cm" (unit is mm) *)
option=Tinkerforge`BrickletDistanceIR`THRESHOLDUOPTIONUSMALLER
dir@SetDistanceCallbackThreshold[option,200,0]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[dir]
ReleaseNETObject[ipcon]
